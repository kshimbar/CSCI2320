Copy Assignment
Vector& Vector::operator = (const Vector& a){
    double* p = new double[a.sz]
    for(int i = 0; i != a.sz; ++i){
        p[i] = a.elem[i]
    }
    delete[] elem;
    elem = p;
    sz = a.sz
    return *this
}
