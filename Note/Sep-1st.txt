We will use "new" to make a variable of array.

Stack.h
class Stack {
    Private:
        int* elem;
        int sz;
        int top;
    Public:
        void Push (int i);
        int pop();
        bool isEmpty();
        int peek();
        Stack(int s){
            elem {new int[s]};
            sz{s};
            top{0};
        }
}

Stack.cpp
void Stack::push(int i){
    if(sz == top){
        //make a bigger array
        int* arr = new int[sz*2];
        for(int t = 0; t < sz; t++){
            arr[t] = elem[t]; 
        }
        delete[] elem;
        elem = arr;
        //if you delete arr, elem can't access to the arr.
        sz = sz * 2;
    }
    elem[top] = i;
    top++;
}
